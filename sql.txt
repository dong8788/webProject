-- 주문상세내역 SQL
CREATE TABLE PURCHASEORDER(
	PONUM NUMBER PRIMARY KEY,
	CUSID VARCHAR2(20),
	PRODSEQ VARCHAR2(20),
	POQUANTITY NUMBER,
	PAYNUM NUMBER,
	RECIPIENT VARCHAR2(20),
	RECIPIENTPHONE VARCHAR2(20),
	RECIPIENTADDR VARCHAR2(20),
	COURIER VARCHAR2(20),
	SHIPPINGNUM VARCHAR2(20),
	POSTAT VARCHAR2(20),
	BILLINGAMOUNT NUMBER,
	ORORDERDATE DATE DEFAULT SYSDATE,
	CONFIRMDATE DATE,
	SHIPDATE DATE,
	RECIPIENTDATE DATE,
	FCONFIRMDATE DATE,
	REJECTDATE DATE,
	PAYMETHOD VARCHAR2(10)
);
-- 주문상세내역 SEQUENCE
CREATE SEQUENCE PONUM_SEQ
INCREMENT BY 1
START WITH 1
MINVALUE 1
NOMAXVALUE
NOCYCLE;

-- 공지사항 FAQ 장바구니 테이블
create sequence "NOTICE_SEQ" start with 1 increment by 1 maxvalue 99999;
create sequence "FAQ_SEQ" start with 1 increment by 1 maxvalue 99999;
create sequence "CART_SEQ" start with 1 increment by 1 maxvalue 99999;

create table "NOTICE" (
    "NOTICESEQ" number primary key,
    "NOTICECATEGORY" varchar2(10),
    "CONTENT" varchar2(1000),
    "TITLE" varchar2(255),
    "REGDATE" timestamp
);

create table "FAQ" (
    "FAQSEQ" number,
    "FAQCATEGRY" varchar2(10),
    "QUERY" varchar2(1000),
    "REPLY" varchar2(1000),
    "REGDATE" timestamp
);

create table "CART" (
    "CUSSEQ" number,
    "PRODSEQ" varchar2(127),
    "POQUANTITY" number
);

-- 상품평, 상품문의
create sequence "REVIEWSEQ" start with 1 increment by 1 maxvalue 99999;

create table "REVIEW"(
	"REVIEWSEQ"	number primary key,
	"PONUM"		number			NOT NULL,
	"SCORE"		number			NOT NULL,
	"TITLE"		varchar2(50)		NOT NULL,
	"CONTENT"	varchar2(500)		NOT NULL,
	"REGDATE"	DATE DEFAULT sysdate
);


create sequence "QUERYSEQ" start with 1 increment by 1 maxvalue 99999;

create table "QUERY"(
	"QUERYSEQ"	number primary key,
	"PUBLIC"		varchar2(10)		NOT NULL,
	"PRODID"		number			NOT NULL,
	"CUSID"		varchar2(50)		NOT NULL,
	"TITLE"		varchar2(50)		NOT NULL,
	"CONTENT"	varchar2(500)		NOT NULL,
	"REGDATE"	DATE DEFAULT sysdate,
	"REPLY"		varchar2(500)		NOT NULL
);



--- 고객센터 문의
create table CENQUERY(
OPEN varchar2(20),
QUERYTYPE varchar2(50),
CUSID varchar2(20),
CONTENT varchar2(1000),
TITLE varchar2(100),
REGDATE date default sysdate,
REPLY varchar2(1000)
);

create sequence "CENQUERYSEQ" start with 1 increment by 1 maxvalue 99999; 


--- 제품정보
create table PRODUCT(
CATEGORY varchar2(50),
PRODNAME varchar2(100),
PRODINFORM varchar2(1000),
PRICE number notnull,
DISCOUNT number,
STOCK NUMBER DEFAULT 9999,
SHIPPINGCHARGE NUMBER DEFAULT 0,
REGDATE date default sysdate,
UPDATEDATE date default sysdate
);

create sequence "PRODSEQ" start with 1 increment by 1 maxvalue 99999; 


---판매자 정보
create sequence "SELLERSEQ" start with 1 increment by 1 maxvalue 99999;

create table "SELLER"(
	"SELLERSEQ"	number primary key,
	"BUSINESSNAME"	varchar2(100) UNIQUE,
	"NAME" 		varchar2(50),
	"PHONE"		varchar2(11),
	"ADDR"		varchar2(50),
	"EMAIL"		varchar2(20) UNIQUE,
	"ID"		varchar2(50) UNIQUE,
	"PASSWORD"	varchar2(50),
	"JOINDATE"	date default sysdate,
	"BUSINESSNUM"	nvarchar2(100) UNIQUE
);
