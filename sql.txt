-- 주문상세내역 SQL
CREATE TABLE PURCHASEORDER(
	PONUM NUMBER PRIMARY KEY,
	CUSID VARCHAR2(20),
	PRODSEQ VARCHAR2(20),
	POQUANTITY NUMBER,
	PAYNUM NUMBER,
	RECIPIENT VARCHAR2(20),
	RECIPIENTPHONE VARCHAR2(20),
	RECIPIENTADDR VARCHAR2(20),
	COURIER VARCHAR2(20),
	SHIPPINGNUM VARCHAR2(20),
	POSTAT VARCHAR2(20),
	BILLINGAMOUNT NUMBER,
	ORORDERDATE DATE DEFAULT SYSDATE,
	CONFIRMDATE DATE,
	SHIPDATE DATE,
	RECIPIENTDATE DATE,
	FCONFIRMDATE DATE,
	REJECTDATE DATE,
	PAYMETHOD VARCHAR2(10)
);
-- 주문상세내역 SEQUENCE
CREATE SEQUENCE PONUM_SEQ
INCREMENT BY 1
START WITH 1
MINVALUE 1
NOMAXVALUE
NOCYCLE;
